## 创建命令文件

### 1. 创建文件

```bash
touch cmd/lc/feature.go
```

### 2. 文件结构模板

```go
package cmd

import (
    "fmt"

    "github.com/spf13/cobra"
    "github.com/user/lc/internal/api"
    "github.com/user/lc/internal/common"
    "go.uber.org/zap"
)

// 包级变量
var (
    featureWorkspaceKey string
    featureLimit        int
)

// 父命令
var featureCmd = &cobra.Command{
    Use:   "feature",
    Short: "管理项目特性",
    Long:  `创建、查询、管理项目特性。`,
}

// 子命令：list
var featureListCmd = &cobra.Command{
    Use:   "list",
    Short: "列出特性",
    Long:  `列出指定研发空间的所有特性。`,
    Run:   runFeatureList,
}

// 子命令：create
var featureCreateCmd = &cobra.Command{
    Use:   "create [name]",
    Short: "创建特性",
    Long:  `创建新的项目特性。`,
    Args:  cobra.ExactArgs(1),
    Run:   runFeatureCreate,
}

// 初始化函数
func init() {
    // 注册子命令
    featureCmd.AddCommand(featureListCmd)
    featureCmd.AddCommand(featureCreateCmd)

    // 添加标志
    featureListCmd.Flags().StringVarP(&featureWorkspaceKey, "workspace-key", "w", "", "研发空间 Key")
    featureListCmd.Flags().IntVarP(&featureLimit, "limit", "l", 10, "返回数量限制")

    featureCreateCmd.Flags().StringVarP(&featureWorkspaceKey, "workspace-key", "w", "", "研发空间 Key")
    featureCreateCmd.Flags().String("description", "", "特性描述")

    // 注册到根命令
    rootCmd.AddCommand(featureCmd)
}
```

### 3. 实现 list 子命令

```go
func runFeatureList(cmd *cobra.Command, args []string) {
    // 1. 尝试自动探测
    workspaceKey := featureWorkspaceKey

    if workspaceKey == "" {
        autoResult := common.TryAutoDetect(true)
        if autoResult.Success {
            workspaceKey = autoResult.Context.WorkspaceKey
            if debugMode {
                common.PrintAutoDetectInfo(autoResult.Context, logger)
            }
        } else {
            common.PrintAutoDetectError(autoResult.Error)
            os.Exit(1)
        }
    }

    // 2. 执行命令
    common.Execute(func(ctx *common.CommandContext) (interface{}, error) {
        // 3. 调用 API
        features, err := api.ListFeatures(ctx, workspaceKey, featureLimit)
        if err != nil {
            return nil, fmt.Errorf("查询特性列表失败: %w", err)
        }

        return map[string]interface{}{
            "items": features,
            "count": len(features),
        }, nil
    }, common.ExecuteOptions{
        DebugMode: debugMode,
        Insecure:  insecureSkipVerify,
        DryRun:    dryRunMode,
        Logger:    &logger,
    })
}
```

### 4. 实现 create 子命令

```go
func runFeatureCreate(cmd *cobra.Command, args []string) {
    // 1. 获取参数
    name := args[0]
    description, _ := cmd.Flags().GetString("description")

    // 2. 尝试自动探测
    workspaceKey := featureWorkspaceKey

    if workspaceKey == "" {
        autoResult := common.TryAutoDetect(true)
        if autoResult.Success {
            workspaceKey = autoResult.Context.WorkspaceKey
        } else {
            common.PrintAutoDetectError(autoResult.Error)
            os.Exit(1)
        }
    }

    // 3. 执行命令
    common.Execute(func(ctx *common.CommandContext) (interface{}, error) {
        // 4. 构建请求
        req := api.CreateFeatureRequest{
            Name:        name,
            Description: description,
            WorkspaceKey: workspaceKey,
        }

        // 5. 调用 API
        result, err := api.CreateFeature(ctx, req)
        if err != nil {
            return nil, fmt.Errorf("创建特性失败: %w", err)
        }

        return result, nil
    }, common.ExecuteOptions{
        DebugMode: debugMode,
        Insecure:  insecureSkipVerify,
        Logger:    &logger,
    })
}
```

## 实现 API 客户端

### 1. 创建 API 文件

```bash
touch internal/api/feature.go
```

### 2. 定义请求/响应结构

```go
package api

// CreateFeatureRequest 创建特性请求
type CreateFeatureRequest struct {
    Name         string `json:"name"`
    Description  string `json:"description,omitempty"`
    WorkspaceKey string `json:"workspaceKey"`
}

// FeatureResponse 特性响应
type FeatureResponse struct {
    ID          string `json:"id"`
    Name        string `json:"name"`
    Description string `json:"description"`
    Status      string `json:"status"`
    CreatedAt   string `json:"createdAt"`
}
```

### 3. 实现 API 方法

```go
// ListFeatures 查询特性列表
func ListFeatures(ctx interface{ GetHeaders(string) map[string]string },
    workspaceKey string, limit int) ([]FeatureResponse, error) {

    // 1. 获取配置
    cfg := ctx.(*common.CommandContext).Config
    client := ctx.(*common.CommandContext).Client

    // 2. 构建请求
    url := fmt.Sprintf("%s/api/features?limit=%d",
        cfg.API.BaseReqURL, limit)

    req := &Request{
        URL:     url,
        Method:  "GET",
        Headers: ctx.GetHeaders(workspaceKey),
    }

    // 3. 发送请求
    resp, err := client.Send(req)
    if err != nil {
        return nil, err
    }
    defer resp.Body.Close()

    // 4. 解析响应
    var result struct {
        Success bool              `json:"success"`
        Data    []FeatureResponse `json:"data"`
        Message string            `json:"message"`
    }

    if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
        return nil, fmt.Errorf("解析响应失败: %w", err)
    }

    if !result.Success {
        return nil, fmt.Errorf("API 错误: %s", result.Message)
    }

    return result.Data, nil
}

// CreateFeature 创建特性
func CreateFeature(ctx *common.CommandContext,
    req CreateFeatureRequest) (*FeatureResponse, error) {

    // 1. 构建请求
    url := fmt.Sprintf("%s/api/features", ctx.Config.API.BaseReqURL)

    apiReq := &Request{
        URL:     url,
        Method:  "POST",
        Headers: ctx.GetHeaders(req.WorkspaceKey),
        Body:    req,
    }

    // 2. 发送请求
    resp, err := ctx.Client.Send(apiReq)
    if err != nil {
        return nil, err
    }
    defer resp.Body.Close()

    // 3. 检查状态码
    if resp.StatusCode != http.StatusCreated {
        body, _ := io.ReadAll(resp.Body)
        return nil, fmt.Errorf("创建失败 (HTTP %d): %s",
            resp.StatusCode, string(body))
    }

    // 4. 解析响应
    var result struct {
        Success bool            `json:"success"`
        Data    FeatureResponse `json:"data"`
    }

    if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
        return nil, fmt.Errorf("解析响应失败: %w", err)
    }

    return &result.Data, nil
}
```

```go
package cmd

import (
    "testing"
)

func TestFeatureListFlags(t *testing.T) {
    // 测试标志解析
    cmd := featureListCmd

    // 检查必需标志
    if cmd.Flag("workspace-key") == nil {
        t.Error("缺少 workspace-key 标志")
    }

    // 检查默认值
    if cmd.Flag("limit").DefValue != "10" {
        t.Errorf("limit 默认值错误: %s", cmd.Flag("limit").DefValue)
    }
}

func TestFeatureCreateArgs(t *testing.T) {
    cmd := featureCreateCmd

    // 测试参数验证
    if cmd.Args == nil {
        t.Error("未设置 Args 验证")
    }

    // 测试有效参数
    err := cmd.Args(cmd, []string{"feature-name"})
    if err != nil {
        t.Errorf("有效参数验证失败: %v", err)
    }

    // 测试无效参数（缺少名称）
    err = cmd.Args(cmd, []string{})
    if err == nil {
        t.Error("缺少参数应该返回错误")
    }
}
```

## 验证和调试

### 1. 编译检查

```bash
# 编译项目
go build ./...

# 检查是否有编译错误
go vet ./...
```

### 2. 本地测试

```bash
# 构建二进制文件
make build

# 测试帮助信息
./bin/lc feature --help
./bin/lc feature list --help
./bin/lc feature create --help

# 测试命令（调试模式）
./bin/lc feature list -w XXJSxiaobaice -d
```

### 3. 调试技巧

#### 使用 Debug 模式

```bash
./bin/lc feature list -w XXJSxiaobaice -d
```

#### 检查自动探测

```bash
./bin/lc detect
```

#### 查看请求详情

在 `api/feature.go` 中添加日志：

```go
ctx.Logger.Debug("Sending request",
    zap.String("url", url),
    zap.String("method", "GET"),
)
```

### 4. 常见问题

| 问题 | 解决方案 |
|------|----------|
| 命令未显示在帮助中 | 检查 `init()` 是否调用了 `rootCmd.AddCommand()` |
| 标志值获取为空 | 检查变量是否绑定到指针，使用 `StringVarP` 而非 `StringP` |
| API 返回 401 | 检查 `config.json` 中的 cookie 是否有效 |
| 自动探测失败 | 确保在 Git 仓库目录下执行，或手动指定 `-w` |
