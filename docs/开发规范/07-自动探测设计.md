## 设计目标

自动探测功能旨在解决以下问题：

1. **减少重复输入**：用户无需每次手动指定 `--workspace-key`
2. **降低使用门槛**：新用户无需了解复杂的参数含义
3. **防止误操作**：自动匹配当前 Git 仓库，避免操作错误项目
4. **提升效率**：一次探测，多次复用

type AutoDetectContext struct {

type AutoDetectResult struct {
    Success bool                // 是否成功
    Context *AutoDetectContext  // 上下文信息
    Error   error               // 错误信息

var (

// ClearAutoDetectCache 清除自动探测缓存
func ClearAutoDetectCache() {
    cachedDetectMu.Lock()
    cachedDetectResult = nil
    cachedDetectMu.Unlock()
}

### 2. 内部执行流程

```go
// performDetect 执行实际的探测逻辑
func performDetect() (*AutoDetectContext, error) {
    ctx := &AutoDetectContext{
        DetectTime: time.Now(),
    }

    // Step 1: 探测 Git 信息
    gitInfo, err := probeGitInfo(".")
    if err != nil {
        ctx.Matched = false
        ctx.MatchReason = fmt.Sprintf("Git 探测失败: %v", err)
        return ctx, nil
    }
    ctx.GitInfo = gitInfo

    if !gitInfo.IsGitRepo {
        ctx.Matched = false
        ctx.MatchReason = "当前目录不在 Git 仓库中"
        return ctx, nil
    }

    // Step 2: 执行 lc detect 命令
    detectResult, err := executeLCDetect()
    if err != nil {
        ctx.Matched = false
        ctx.MatchReason = fmt.Sprintf("lc detect 执行失败: %v", err)
        return ctx, nil
    }

    // Step 3: 解析结果
    if matched, ok := detectResult["matched"].(bool); !ok || !matched {
        ctx.Matched = false
        ctx.MatchReason = "未能匹配到远程仓库"
        return ctx, nil
    }

    // Step 4: 提取关键字段
    ctx.Matched = true
    ctx.WorkspaceKey = detectResult["workspaceKey"].(string)
    ctx.WorkspaceName = detectResult["workspaceName"].(string)
    // ... 提取其他字段

    return ctx, nil
}
```

### 3. Git 探测实现

```go
// probeGitInfo 探测 Git 仓库信息
func probeGitInfo(path string) (*GitInfo, error) {
    absPath, err := filepath.Abs(path)
    if err != nil {
        return nil, err
    }

    info := &GitInfo{
        CurrentPath: absPath,
    }

    // 检查是否在 Git 仓库中
    cmd := exec.Command("git", "-C", absPath, "rev-parse", "--git-dir")
    if _, err := cmd.Output(); err != nil {
        info.IsGitRepo = false
        return info, nil
    }

    info.IsGitRepo = true

    // 获取仓库名称
    cmd = exec.Command("git", "-C", absPath, "rev-parse", "--show-toplevel")
    if output, err := cmd.Output(); err == nil {
        info.RepoName = filepath.Base(strings.TrimSpace(string(output)))
    }

    // 获取 remote URL
    cmd = exec.Command("git", "-C", absPath, "remote", "get-url", "origin")
    if output, err := cmd.Output(); err == nil {
        info.RemoteURL = strings.TrimSpace(string(output))
    }

    return info, nil
}
```

### 4. lc detect 命令调用

```go
// executeLCDetect 执行 lc detect 命令获取上下文
func executeLCDetect() (map[string]interface{}, error) {
    // 1. 查找 lc 命令
    lcPath, err := exec.LookPath("lc")
    if err != nil {
        // 尝试使用 ./bin/lc
        if _, err := os.Stat("./bin/lc"); err == nil {
            lcPath = "./bin/lc"
        } else {
            return nil, fmt.Errorf("lc 命令未找到: %w", err)
        }
    }

    // 2. 执行命令
    cmd := exec.Command(lcPath, "detect", "-k")
    output, err := cmd.Output()
    if err != nil {
        if exitErr, ok := err.(*exec.ExitError); ok {
            return nil, fmt.Errorf("lc detect 执行失败: %s", string(exitErr.Stderr))
        }
        return nil, fmt.Errorf("lc detect 执行失败: %w", err)
    }

    // 3. 解析 JSON 输出
    var result map[string]interface{}
    if err := json.Unmarshal(output, &result); err != nil {
        return nil, fmt.Errorf("解析 lc detect 输出失败: %w", err)
    }

    // 4. 提取 data 字段
    if data, ok := result["data"].(map[string]interface{}); ok {
        return data, nil
    }

    return result, nil
}
```

## 缓存机制

### 1. 缓存策略

```go
func GetAutoDetectContext() (*AutoDetectContext, error) {
    // 1. 先检查缓存
    cachedDetectMu.RLock()
    if cachedDetectResult != nil &&
       time.Since(cachedDetectResult.DetectTime) < cacheValidDuration {
        result := cachedDetectResult
        cachedDetectMu.RUnlock()
        return result, nil
    }
    cachedDetectMu.RUnlock()

    // 2. 执行探测
    ctx, err := performDetect()
    if err != nil {
        return nil, err
    }

    // 3. 更新缓存
    cachedDetectMu.Lock()
    cachedDetectResult = ctx
    cachedDetectMu.Unlock()

    return ctx, nil
}
```

## 复用指南

### 1. 命令中启用自动探测

```go
// 在命令的 Run 函数中
var reqListCmd = &cobra.Command{
    Use:   "list",
    Run: func(cmd *cobra.Command, args []string) {
        // 1. 获取显式参数
        workspaceKey, _ := cmd.Flags().GetString("workspace-key")

        // 2. 如果未指定，尝试自动探测
        if workspaceKey == "" {
            autoResult := common.TryAutoDetect(true)
            if autoResult.Success {
                workspaceKey = autoResult.Context.WorkspaceKey

                // 可选：打印探测信息
                if debugMode {
                    common.PrintAutoDetectInfo(autoResult.Context, logger)
                }
            } else {
                // 探测失败，打印帮助信息并退出
                common.PrintAutoDetectError(autoResult.Error)
                os.Exit(1)
            }
        }

        // 3. 执行命令
        common.Execute(func(ctx *common.CommandContext) (interface{}, error) {
            // 使用 workspaceKey 执行操作...
            return api.ListRequirements(ctx, workspaceKey)
        }, opts)
    },
}
```

## 注意事项

1. **性能考虑**：自动探测会执行外部命令，5 分钟缓存可降低开销
2. **并发安全**：缓存使用 RWMutex 保证并发安全
3. **错误处理**：探测失败不是致命错误，应优雅降级
4. **用户提示**：自动探测成功/失败都应给用户明确反馈
